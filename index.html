// Add vector lines background animation const canvas = document.createElement('canvas'); canvas.width = window.innerWidth; canvas.height = window.innerHeight; canvas.style.position = 'fixed'; canvas.style.top = '0'; canvas.style.left = '0'; canvas.style.zIndex = '-1'; document.body.appendChild(canvas); const context = canvas.getContext('2d'); context.strokeStyle = 'rgba(255, 255, 255, 0.3)'; context.lineWidth = 1; const numLines = 50; const lines = []; for (let i = 0; i < numLines; i++) { lines.push({ x: Math.random() * window.innerWidth, y: Math.random() * window.innerHeight, length: Math.random() * 100, angle: Math.random() * Math.PI * 2, }); } function draw() { context.clearRect(0, 0, window.innerWidth, window.innerHeight); for (let i = 0; i < numLines; i++) { const line = lines[i]; const endX = line.x + Math.cos(line.angle) * line.length; const endY = line.y + Math.sin(line.angle) * line.length; context.beginPath(); context.moveTo(line.x, line.y); context.lineTo(endX, endY); context.stroke(); line.x = endX; line.y = endY; if (line.x < 0 || line.x > window.innerWidth) { line.angle = Math.PI - line.angle; } if (line.y < 0 || line.y > window.innerHeight) { line.angle = -line.angle; } } requestAnimationFrame(draw); } requestAnimationFrame(draw);
